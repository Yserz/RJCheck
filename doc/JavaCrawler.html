def crawl
        self.chdir

        Dir.glob("**/*.java") do |absolute_file_path|
                puts @path
                puts File.basename(absolute_file_path)
                puts absolute_file_path
                next if File.directory? absolute_file_path
                data = IO.read(absolute_file_path)
                yield absolute_file_path, data
        end
end
